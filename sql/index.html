function order(megaOBJ) {
    var desc = megaOBJ.sort((a, b) => b.numberofTimes - a.numberofTimes); //desc order on nooftimes
    var mid = Math.floor(megaOBJ.length / 2);
    var len = megaOBJ.length;
    let temp = new Array(len);
    var tempL = []
    var tempR = []
    temp[mid] = desc[0];
    for (let index = 1; index < len; index += 2) {
        tempL.push(desc[index]);
    }
    for (let index = 2; index < len; index += 2) {
        tempR.push(desc[index]);
    }
    for (let index = 0; index < tempR.length; index++) {
        temp[mid + index + 1] = tempR[index];
    }
    for (let index = 0; index < tempL.length; index++) {
        temp[mid - index - 1] = tempL[index];
    }
    //final
    for (let index = 0; index < temp.length; index++) {
        temp[index].cx = index + 1;
        temp[index].cy = temp[index].cYear;
    }

    
    return temp;
