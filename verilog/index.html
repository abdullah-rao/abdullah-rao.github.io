VHDL/Hardware Modeling

These are exerpts from the computer organization course in which we implemented basic computer architecture using individual hardware building blocks. Verilog was used to model each component's wiring and logic. These individual modules were then reused to construct more complex components such as multiplexers. The end result was a basic computer circuit that was capable of reading and executing program instructions using program counter and corresponding hardware parts.

module LabM;
   reg [31:0] PCin;
   reg RegDst, RegWrite, clk, ALUSrc, MemRead, MemWrite, Mem2Reg;
   reg [2:0] op;
   wire [31:0] wd, rd1, rd2, imm, ins, PCp4, z, memOut, wb;
   wire [25:0] jTarget;
   wire zero;
   
   yIF myIF(ins, PCp4, PCin, clk);
   yID myID(rd1, rd2, imm, jTarget, ins, wd, RegDst, RegWrite, clk);
   yEX myEX(z, zero, rd1, rd2, imm, op, ALUSrc);
   yDM myDM(memOut, z, rd2, clk, MemRead, MemWrite);
   yWB myWB(wb, z, memOut, Mem2Reg);
   assign wd = wb;
   
   initial
      begin
         PCin = 128; clk = 0;
         repeat (11)
         begin
            clk = 1; #1;
            op = 3'b010;
            RegDst = 0; RegWrite = 0; ALUSrc = 1;
            MemRead = 0; MemWrite = 0; Mem2Reg = 0;
            if (ins[31:26] == 0)
               begin
                  RegDst = 1; RegWrite = 1; ALUSrc = 0; // All R-type ins
                  if (ins[5:0] == 37)
                     op = 3'b001; // intruction or
               end
            if (ins[31:26] == 4)
               begin
                  ALUSrc = 0; op = 3'b110; // intruction beq
               end
            if (ins[31:26] == 35)
               begin
                  RegWrite = 1; MemRead = 1; Mem2Reg = 1; // intruction lw
               end
            if (ins[31:26] == 43)
               MemWrite = 1; // inrtuction sw
            if (ins[31:26] == 8)
               RegWrite = 1; // intruction addi
            #1; clk = 0;
            $display("%h: rd1=%2d rd2=%2d z=%3d zero=%b wb=%2d",
               ins, rd1, rd2, z, zero, wb);
            //---------------------------------Prepare for the next ins
		if ((ins[31:26] == 4 && zero) == 1)
			PCin = PCp4 + (imm << 2) ;//shifted left twice;
		else if (ins[31:26] == 2)
			PCin = jTarget << 2;// shifted left twice;
		else
			PCin = PCp4;
         end
         $finish;
      end
endmodule
